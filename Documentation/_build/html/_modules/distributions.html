        upper_border = self.mu+(3*self.sigma)

[docs]class Distribution_mixture_of_normals_truncated_custom(IDistributions): # mus of mixture need to be within the borders
    def __init__(self, list_of_mus:List[Type[float]], sigma:float, lower_border=0.0, upper_border=1.0):
        self.sigma = sigma
        self.lower_border = lower_border
        self.upper_border = upper_border
        for mu in list_of_mus:
            if not (lower_border-2*sigma <= mu <= upper_border+2*sigma):
                raise ValueError("All values in list_of_mus must be between lower_border and upper_border")
        self.list_of_mus = list_of_mus
