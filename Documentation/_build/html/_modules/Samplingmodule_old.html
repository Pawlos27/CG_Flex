    new_nodelist = []
    if len(shuffle_list) >0 :
        for id in shuffle_list:
            for node in nodelist:
                if node.id == id:
                    new_nodelist.append(node)
                    nodelist.remove(node)
    else:
        new_nodelist=nodelist
    return new_nodelist

[docs]def calculate_values_of_dependencies_from_specific_id_downwards(nodelist: List[Type[Nodeobject]]  ,start_id:int): 
    nodelist = nodelist
    calculation_starter = False
    starting_index = None
    while calculation_starter == False:
            for j in range(len(nodelist)):
                if nodelist[j].id == start_id:
                    calculation_starter = True
                    starting_index=j
                    break
    if calculation_starter == True :
        for i in range(starting_index,len(nodelist)):
            if isinstance(nodelist[i].dependency, (float, int)):
                nodelist[i].value = nodelist[i].dependency
            elif isinstance(nodelist[i].dependency, distributions.IDistributions):
                nodelist[i].value = nodelist[i].dependency.get_value_from_distribution()
            elif isinstance(nodelist[i].dependency, _dependencymaker.Dependencies):
                values_parents = get_values_parents_from_nodelist(parents=nodelist[i], nodelist=nodelist)
                nodelist[i].value= nodelist[i].dependency.calculate_normalized_value(x_values=values_parents)
    elif calculation_starter == False :
            raise ValueError("id not found in nodelist")
    return  nodelist

        for node in self.nodelist:
            if isinstance(node.dependency, (float, int)):
                node.value = node.dependency
            elif isinstance(node.dependency, distributions.IDistributions):
                node.value = node.dependency.get_value_from_distribution()
            elif isinstance(node.dependency, _dependencymaker.Dependencies):
                parents = node.parents
                values_parents = get_values_parents_from_nodelist(parents=parents, nodelist=self.nodelist)
                node.value= node.dependency.calculate_normalized_value(x_values=values_parents)
            if node.value > 1:
                node.value = 1
                print(f"wertüberschreitung an node {node.id}")
            if node.value < 0:
                node.value = 0
                print(f"wertunterschreitung an node {node.id}")

[docs]    def make_accumulated_samples(self): # sorting all sampled values to the linked node_id 

        accumulated_list = []
        component_lenght = len(self.id_value_list_of_arrays[0])
        numer_of_samples = len(self.id_value_list_of_arrays)
        for i in range(component_lenght):
            acumulated_list_per_node_container = []
            for j in range(numer_of_samples):

            accumulated_list_per_node = [i, acumulated_list_per_node_container]
            accumulated_list.append(accumulated_list_per_node)
        accumulated_list.append(accumulated_list)
        self.samples_accumulated_per_id = accumulated_list
