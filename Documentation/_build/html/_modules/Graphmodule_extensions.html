[docs]    def show_graph(self, plot_title="DAG_Graph"):
        positions_f= self._make_positions_dictionary()
        G = nx.DiGraph()
        G.add_edges_from(self.verticelist)
        nx.draw_networkx(G, positions_f ,with_labels=True)
        plt.xlabel("Layer Axis")
        plt.ylabel("Thorus Axis")
        plt.title(plot_title)
        plt.show()

        
    def _make_positions_dictionary(self):
        nodelist = self.nodelist
        positions={}
        nodelist.reverse()
        for node in nodelist:
            positions[node.id]=[node.coord_l,node.coord_t[0]]
        return positions
    

[docs]    def make_verticelist(self):
        verticelist=[]
        for node in self.nodelist:
            for j in node.children:
                kante = (node.id, j)
                verticelist.append(kante)
        return verticelist
