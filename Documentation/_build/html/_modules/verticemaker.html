[docs]def edge_probability_calc(t_distance:float, l_distance:float, edgeprobability_thorus:Iedgeprobability, edgeprobability_layer:Iedgeprobability, c_faktor: float ):
    edge_prob_t= edgeprobability_thorus.probability_calc(distance=t_distance)
    edge_prob_l= edgeprobability_layer.probability_calc(distance=l_distance)
    c_f= c_faktor
    edge_probability_f=  (c_f*edge_prob_l*edge_prob_t)**2
    if edge_probability_f > 1:
            edge_probability_f  = 1
    return edge_probability_f




[docs]def parameter_c_calc(nodelist:list,number_of_edges:int,edgeprobability_layer: Iedgeprobability,edgeprobability_thorus: Iedgeprobability) :
    #HIER Berechnen expected value
    number_of_edges_f=number_of_edges
    nodelist_f= nodelist
    listlenght= len(nodelist_f)
    scale_val=nodelist_f[1].scale_val
    c_value = 1
    lower_c_limit=0
    upper_c_limit= None
    expected_edges = 0
    while expected_edges  < 0.8*number_of_edges_f or expected_edges > 1.2*number_of_edges_f :
        n = 0
        expected_edges = 0
        for node in nodelist:
            for j in range(n+1,listlenght) :
                if nodelist[n].sink == False and nodelist[j].source == False and nodelist_f[n].layer != nodelist_f[j].layer and nodelist_f[n].layer != nodelist_f[j].layer:
                    coord_t_parent_f= nodelist_f[n].coord_t 
                    coord_t_child_f= nodelist_f[j].coord_t 
                    coord_l_parent_f=nodelist_f[n].coord_l
                    coord_l_child_f=nodelist_f[j].coord_l
                    t_distance_f = edge_distance_calc_thorus(coord_t_parent=coord_t_parent_f, coord_t_child=coord_t_child_f)
                    l_distance_f = edge_distance_calc_layer(coord_l_parent=coord_l_parent_f, coord_l_child=coord_l_child_f)
                    edge_probability =edge_probability_calc(t_distance=t_distance_f, l_distance=l_distance_f,edgeprobability_thorus=edgeprobability_thorus,edgeprobability_layer=edgeprobability_layer, c_faktor= c_value)
                    expected_edges += edge_probability
            n+=1
        if expected_edges  < 0.95*number_of_edges_f:
            lower_c_limit= c_value
            if upper_c_limit ==  None:
                c_value= 2*c_value
            else:
                c_value= (upper_c_limit+lower_c_limit)/2

        elif expected_edges > 1.05*number_of_edges_f:
            upper_c_limit= c_value
            if lower_c_limit == 0:
                c_value= c_value/2
            else:
                c_value= (upper_c_limit+lower_c_limit)/2
    return c_value

[docs]    def make_edges(self,nodelist: List[Nodeobject])-> List[Nodeobject]:
        layerlimit = self.layerlimit
        number_of_edges = self.number_of_edges
        edgeprobability_layer = self.edgeprobability_layer
        edgeprobability_thorus = self.edgeprobability_thorus
        edge_outdegree_max = self.edge_outdegree_max
        edge_indegree_max = self.edge_indegree_max
        nodelist_f = nodelist
        listlenght=len(nodelist)
        scale_val=nodelist_f[1].scale_val
        c= parameter_c_calc(nodelist=nodelist_f,edgeprobability_layer=edgeprobability_layer, edgeprobability_thorus=edgeprobability_thorus, number_of_edges=number_of_edges )
        # reset all edges
        for node in nodelist_f:
            node.parents = []
            node.children = []
        # first loop
        n = 0
        for node in nodelist_f:
            for j in range(n+1,listlenght ):
                if nodelist_f[n].sink == False and nodelist_f[j].source == False and len(nodelist_f[n].children) <edge_outdegree_max and len(nodelist_f[j].parents) < edge_indegree_max and nodelist_f[j].layer - nodelist_f[n].layer < layerlimit and nodelist_f[n].layer != nodelist_f[j].layer:
                    coord_t_parent_f=nodelist_f[n].coord_t
                    coord_t_child_f=nodelist_f[j].coord_t 
                    coord_l_parent_f=nodelist_f[n].coord_l
                    coord_l_child_f=nodelist_f[j].coord_l
                    t_distance_f = edge_distance_calc_thorus(coord_t_parent=coord_t_parent_f, coord_t_child=coord_t_child_f)
                    l_distance_f = edge_distance_calc_layer(coord_l_parent=coord_l_parent_f, coord_l_child=coord_l_child_f)
                    edge_probability = edge_probability_calc(t_distance=t_distance_f, l_distance=l_distance_f,edgeprobability_thorus=edgeprobability_thorus,edgeprobability_layer=edgeprobability_layer, c_faktor= c )
                    edgecheck = edgechecker(edge_probability)
                    if edgecheck == True :
                        print("edge found in first round")
                        nodelist_f[n].children.append(nodelist_f[j].id)
                        nodelist_f[j].parents.append(nodelist_f[n].id)
            n+=1
        # checking if every node who is supposed to has outgoin edges
        n=0
        for node in nodelist_f:
            if nodelist_f[n].sink == False and len(nodelist_f[n].children) == 0 :
                counter_1 = 0
                while len(nodelist_f[n].children) < 1 and counter_1<50:
                    for j in range(n+1,listlenght ):
                        if nodelist_f[j].source == False and nodelist_f[j].layer - nodelist_f[n].layer < layerlimit :
                            coord_t_parent_f= nodelist_f[n].coord_t 
                            coord_t_child_f= nodelist_f[j].coord_t 
                            coord_l_parent_f= nodelist_f[n].coord_l
                            coord_l_child_f= nodelist_f[j].coord_l
                            t_distance_f = edge_distance_calc_thorus(coord_t_parent=coord_t_parent_f, coord_t_child=coord_t_child_f)
                            l_distance_f = edge_distance_calc_layer(coord_l_parent=coord_l_parent_f, coord_l_child=coord_l_child_f)
                            edge_probability = edge_probability_calc(t_distance=t_distance_f, l_distance=l_distance_f,edgeprobability_thorus=edgeprobability_thorus,edgeprobability_layer=edgeprobability_layer, c_faktor= c )
                            edgecheck = edgechecker(edge_probability)
                            if edgecheck == True :
                                nodelist_f[n].children.append(nodelist_f[j].id)
                                nodelist_f[j].parents.append(nodelist_f[n].id)
                                break
                    counter_1 += 1 

                    print(f"Verticcemaker: the first correction loop round:   {counter_1}")
                if len(nodelist_f[n].children) == 0:
                    node.sink = True
            n+=1
    
        # checking if every node who is supposed to, has ingoin edges
        nodelist_f.reverse()
        n=0
        
        for node in nodelist_f:
            if nodelist_f[n].source == False and len(nodelist_f[n].parents) == 0 :
                counter_2 = 0 
                while len(nodelist_f[n].parents) < 1 and counter_2<50:
                    for j in range(n+1,listlenght ):
                        if nodelist_f[j].sink == False and  nodelist_f[n].layer - nodelist_f[j].layer  < layerlimit :
                            coord_t_parent_f= nodelist_f[j].coord_t 
                            coord_t_child_f= nodelist_f[n].coord_t 
                            coord_l_parent_f=nodelist_f[j].coord_l
                            coord_l_child_f=nodelist_f[n].coord_l
                            t_distance_f = edge_distance_calc_thorus(coord_t_parent=coord_t_parent_f, coord_t_child=coord_t_child_f)
                            l_distance_f = edge_distance_calc_layer(coord_l_parent=coord_l_parent_f, coord_l_child=coord_l_child_f)
                            edge_probability = edge_probability_calc(t_distance=t_distance_f, l_distance=l_distance_f,edgeprobability_thorus=edgeprobability_thorus,edgeprobability_layer=edgeprobability_layer, c_faktor= c )
                            edgecheck = edgechecker(edge_probability)
                            if edgecheck == True :
                                nodelist_f[j].children.append(nodelist_f[n].id)
                                nodelist_f[n].parents.append(nodelist_f[j].id)
                                break
                    counter_2 += 1 
                    print(f"Verticcemaker: the second correction loop round:   {counter_2}")
                if len(nodelist_f[n].parents) == 0:
                    node.source = True               
            n+=1
        nodelist_f.reverse()
        print(f"c value is {c}")
