[docs]def vertice_probability_calc(t_distance, l_distance, verticeprobability_thorus=IVerticeprobability, verticeprobability_layer=IVerticeprobability ):
    vertice_prob_t= verticeprobability_thorus().probability_calc(t_distance)
    vertice_prob_l= verticeprobability_layer().probability_calc(l_distance)
    vertice_probability_f= vertice_prob_l*vertice_prob_t
    return vertice_probability_f

    nodelist_f= nodelist

    scale_val=nodelist_f[1].scale_val
    expected_vertices = 0
    n = 0
    for node in nodelist:
        counter = 0
        for j in range(n+1,listlenght) :

                downscaled_coord_t_parent_f= [i/scale_val for i in nodelist_f[n].coord_t ]
                downscaled_coord_t_child_f=[i/scale_val for i in nodelist_f[j].coord_t ]
                downscaled_coord_l_parent_f=nodelist_f[n].coord_l/scale_val
                downscaled_coord_l_child_f=nodelist_f[j].coord_l/scale_val
                t_distance_f = vertice_distance_calc_thorus(coord_t_parent=downscaled_coord_t_parent_f, coord_t_child=downscaled_coord_t_child_f)
                l_distance_f = vertice_distance_calc_layer(coord_l_parent=downscaled_coord_l_parent_f, coord_l_child=downscaled_coord_l_child_f)
                vertice_probability =vertice_probability_calc(t_distance=t_distance_f, l_distance=l_distance_f,verticeprobability_thorus=verticeprobability_thorus,verticeprobability_layer=verticeprobability_layer)
              

                expected_vertices += vertice_probability
        n+=1
    c_value = number_of_vertices/expected_vertices
    return c_value
