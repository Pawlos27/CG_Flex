[docs]    def make_graph(self):
        """this method uses the functionality of the class objects contained in the config object to create the graph
        """
        nodelist_raw = self.inputs.nodemaker.make_nodelist()
        nodelist_layered = self.inputs.layermaker.make_layer(nodelist= nodelist_raw)
        nodelist_sourced = self.inputs.sourcemaker.make_sources(nodelist= nodelist_layered )
        nodelist_sinked = self.inputs.sinkmaker.make_sinks(nodelist= nodelist_sourced)
        self.nodelist = self.inputs.edgemaker.make_edges(nodelist=nodelist_sinked)

        self.nodelist = self.inputs.edgemaker.make_edges(nodelist=nodelist_layered)

[docs]    def safe_nodelist_pikle(self,file_name:str, file_path:Optional[str]):
        if self.nodelist is not None:
            if file_path == None:
                data_folder = os.path.join(os.path.dirname(__file__), 'data')
                file_path = os.path.join(data_folder, file_name)

            with open(file_path, 'wb') as file:
                pickle.dump(self.nodelist, file)
        else:
            raise ValueError("No data to save. Please set data using set_data method first.")

        

[docs]    def load_nodelist_pikle(self,file_name:str,file_path:Optional[str]):
        if file_path == None:
            data_folder = os.path.join(os.path.dirname(__file__), 'data')
            file_path = os.path.join(data_folder, file_name)

        with open(file_path, 'rb') as file:
            self.nodelist = pickle.load(file)

[docs]    def get_list_of_sources(self):
        if self.nodelist == None:
            raise ValueError("there is no graph yet, please make a graph")
        else:
            list_of_sources = []
            for node in self.nodelist:
                if node.source == True:
                    list_of_sources.append(node.id)
            return list_of_sources
